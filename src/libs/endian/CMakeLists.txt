PROJECT (endian)

########################################################################
# Set all source files
########################################################################
SET ( LIB_SRCS
    ${PROJECT_SOURCE_DIR}/amilittle.c
)

########################################################################
# Set include paths
########################################################################
SET ( LIB_INCS
    ${PROJECT_SOURCE_DIR}/include
)

IF ( CFG_GEN_NIOS_TARGET )
    INCLUDE (nios2.cmake)
ENDIF ( CFG_GEN_NIOS_TARGET )

INCLUDE_DIRECTORIES ( ${LIB_INCS} 
                      ${TARGET_DIR}/include
)

########################################################################
# Build library
########################################################################
ADD_LIBRARY ( endian ${LIB_TYPE} ${LIB_SRCS} )
